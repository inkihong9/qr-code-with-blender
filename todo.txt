============ sample Dockerfile =============

FROM ubuntu:22.04

# Install dependencies
RUN apt-get update && apt-get install -y \
    wget \
    libx11-6 libxi6 libxxf86vm1 libxfixes3 libxrender1 libxrandr2 \
    libasound2 libgl1 libegl1 && \
    rm -rf /var/lib/apt/lists/*

# Download official Blender 4.2 Linux release
RUN wget https://download.blender.org/release/Blender4.2/blender-4.2.0-linux-x64.tar.xz && \
    tar -xf blender-4.2.0-linux-x64.tar.xz && \
    mv blender-4.2.0-linux-x64 /opt/blender && \
    ln -s /opt/blender/blender /usr/local/bin/blender

WORKDIR /workspace
ENTRYPOINT ["blender"]
